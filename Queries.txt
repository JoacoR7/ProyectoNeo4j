 - Borrar todos los nodos
match (a)
delete a

- Obtener todos los nodos
match (a)
return a

- Crear nodo
CREATE (:Tipo{
    atributo1: x,
    atributo2: y
})

- Borrar nodo por id
MATCH (n)
WHERE id(n) = 0
DELETE n

- Crear una relación entre nodos
CREATE (n1:Etiqueta1)-[r:Relación]->(n2:Etiqueta2)

MATCH (a), (b)
WHERE a.nombre = "Ana" AND b.nombre = "Juan"
CREATE (a)-[r:Amigo]->(b)

- Cambiar los atributos de un nodo
MATCH (a:Persona {nombre: "Ana"})
SET a.edad = 30

- Borrar relacion
MATCH (n1:Tipo1)-[r:Relacion]-(n2:Tipo2)
WHERE ID(n1) = id1 AND ID(n2) = id2
DELETE r
